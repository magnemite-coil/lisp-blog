SwiftUIで宣言的UIを構築

📅 2025年10月4日 👤 ios_developer ⏱️ 11分で読める

SwiftUIは、宣言的な構文でiOSアプリのUIを構築できる革新的なフレームワークです。 Viewプロトコルとモディファイアを組み合わせて、美しいインターフェースを作成しましょう。

基本的なViewの構造

Swift 
import SwiftUI

struct ContentView: View {
    @State private var counter = 0
    @State private var isShowingDetail = false
    
    var body: some View {
        VStack(spacing: 20) {
            Text("カウンター: \(counter)")
                .font(.title)
                .fontWeight(.bold)
            
            HStack(spacing: 15) {
                Button("減らす") {
                    counter -= 1
                }
                .buttonStyle(.bordered)
                
                Button("増やす") {
                    counter += 1
                }
                .buttonStyle(.borderedProminent)
            }
            
            Button("詳細を表示") {
                isShowingDetail.toggle()
            }
        }
        .padding()
        .sheet(isPresented: $isShowingDetail) {
            DetailView(count: counter)
        }
    }
}

カスタムビューとモディファイア

Swift 
struct DetailView: View {
    let count: Int
    @Environment(\.dismiss) var dismiss
    
    var body: some View {
        NavigationView {
            VStack {
                Image(systemName: "star.fill")
                    .resizable()
                    .scaledToFit()
                    .frame(width: 100, height: 100)
                    .foregroundColor(.yellow)
                
                Text("現在のカウント")
                    .font(.headline)
                
                Text("\(count)")
                    .font(.system(size: 72, weight: .bold))
                    .foregroundColor(.blue)
            }
            .navigationTitle("詳細")
            .toolbar {
                Button("閉じる") {
                    dismiss()
                }
            }
        }
    }
}
