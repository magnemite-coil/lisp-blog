Zigは、C言語の代替を目指す新しいシステムプログラミング言語です。 メモリ安全性、明示的なエラーハンドリング、そしてコンパイル時評価の力を活用しましょう。

基本的な構文とメモリ管理

Zig 
const std = @import("std");

// 構造体の定義
const Point = struct {
    x: f64,
    y: f64,

    pub fn distance(self: Point, other: Point) f64 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return std.math.sqrt(dx * dx + dy * dy);
    }
};

// エラーハンドリング
const DivisionError = error{
    DivisionByZero,
};

fn divide(a: f64, b: f64) DivisionError!f64 {
    if (b == 0) {
        return DivisionError.DivisionByZero;
    }
    return a / b;
}

// メイン関数
pub fn main() !void {
    const p1 = Point{ .x = 0, .y = 0 };
    const p2 = Point{ .x = 3, .y = 4 };
    
    const dist = p1.distance(p2);
    std.debug.print("距離: {d}\n", .{dist});

    // エラーハンドリング
    const result = divide(10, 2) catch |err| {
        std.debug.print("エラー: {}\n", .{err});
        return;
    };
    
    std.debug.print("結果: {d}\n", .{result});
}

アロケータとメモリ管理

const std = @import("std");
const ArrayList = std.ArrayList;

pub fn processNumbers() !void {
    // アロケータの取得
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    
    const allocator = gpa.allocator();

    // 動的配列の作成
    var numbers = ArrayList(i32).init(allocator);
    defer numbers.deinit();

    // 要素の追加
    try numbers.append(1);
    try numbers.append(2);
    try numbers.append(3);

    // スライスの操作
    for (numbers.items) |num| {
        std.debug.print("{d} ", .{num});
    }

    // メモリの明示的な割り当て
    const buffer = try allocator.alloc(u8, 1024);
    defer allocator.free(buffer);

    // バッファの使用
    std.mem.set(u8, buffer, 0);
}

// コンパイル時評価
fn fibonacci(comptime n: u32) u32 {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

const fib_10 = fibonacci(10);  // コンパイル時に計算
