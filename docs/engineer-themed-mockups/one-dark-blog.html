Kotlinは、Javaとの互換性を保ちながら、より簡潔で安全なコードを書けます。 Null安全性やデータクラス、拡張関数など、Kotlinの強力な機能を活用しましょう。

データクラスとNull安全性

Kotlin 
// データクラスの定義
data class User(
    val id: Int,
    val name: String,
    val email: String?,  // Nullable型
    val age: Int = 0      // デフォルト値
)

fun processUser(user: User) {
    // スマートキャストとNull安全呼び出し
    val emailLength = user.email?.length ?: 0
    
    // letスコープ関数
    user.email?.let {
        println("Email: $it")
    }
    
    // whenを使ったパターンマッチング
    when {
        user.age < 18 -> println("未成年")
        user.age in 18..65 -> println("成人")
        else -> println("シニア")
    }
}

拡張関数とコレクション操作

Kotlin 
// 拡張関数の定義
fun String.isValidEmail(): Boolean {
    return contains("@") && contains(".")
}

// コレクション操作
fun filterAndTransformUsers(users: List<User>): List<String> {
    return users
        .filter { it.age >= 18 }
        .filter { it.email?.isValidEmail() == true }
        .map { "${it.name} (${it.email})" }
        .sortedBy { it }
}

// 使用例
val users = listOf(
    User(1, "山田", "yamada@example.com", 25),
    User(2, "佐藤", "sato@example.com", 17),
    User(3, "鈴木", null, 30)
)

val result = filterAndTransformUsers(users)
println(result)
